--KOMENTAR--

CREATE OR REPLACE TRIGGER "KOMENTAR_PK" 
    BEFORE INSERT
    ON KOMENTAR
    REFERENCING NEW AS NEW
    FOR EACH ROW
BEGIN
    SELECT KOMENTAR_ID_SEQ.NEXTVAL INTO :NEW.ID FROM DUAL;
END;
/
ALTER TRIGGER "KOMENTAR_PK" ENABLE;
